var curPfrom = 0;
var curPto = 0;
var isPriceSel = false;
function searchPrice(form){
    var pfrom = $(form).find('input[name="pfrom"]').val();
    var dpfrom = $(form).find('input[name="pfrom"]').attr('dvalue');
    var pto = $(form).find('input[name="pto"]').val();
    var dpto = $(form).find('input[name="pto"]').attr('dvalue');
    if(pfrom == dpfrom && pto == dpto && pfrom.length > 0 && pto.length > 0){
        return false;
    }
    if(pfrom == '' && pto == ''){
        //return false;
    }
    pfrom = (pfrom == '')?0:pfrom;
    pto = (pto == '')?0:pto;
    curPfrom = pfrom;
    curPto = pto;
    isPriceSel = true;
    doSearch();
    return false;
}

function doSearch(key, value) { 
    var urlArr = location.href.split('?');
    var newParams = new Array();
    var params = new Array();
    if (urlArr[1] != undefined) {
        var params = urlArr[1].split('&');
    } 

    var i = 0;
    for (var x =0;x < params.length; x++) {
        var tmp = params[x].split('=');
        if (tmp[0] == 'pfrom' || tmp[0] == 'pto') {
            continue;
        }
        if (tmp[1] != 0 && tmp[0] != key) { 
            newParams[i] = params[x];
            i++; 
        }
    }
    if (value > 0)
        newParams[i] = key+'='+value;

    //加上价格
    if (curPfrom > 0 || curPto > 0) {
        i++;
        newParams[i] = 'pfrom='+curPfrom;
        i++;
        newParams[i] = 'pto='+curPto;
    }else if(!isPriceSel){
        var dvaluePfrom = $('.searchPriceShai').find('input[name="pfrom"]').attr('dvalue');
        curPfrom = (dvaluePfrom != '' && dvaluePfrom != undefined && dvaluePfrom > 0)?parseFloat(dvaluePfrom):0;
        var dvaluePto = $('.searchPriceShai').find('input[name="pto"]').attr('dvalue');
        curPto = (dvaluePto != '' && dvaluePto != undefined && dvaluePto > 0)?parseFloat(dvaluePto):0;
        if(curPfrom > 0 || curPto > 0){
            i++;
            newParams[i] = 'pfrom='+curPfrom;
            i++;
            newParams[i] = 'pto='+curPto;
        }
    }else if(isPriceSel){
        i++;
        newParams[i] = 'pfrom='+curPfrom;
        i++;
        newParams[i] = 'pto='+curPto;
    }

    var curUrl = urlArr[0];
    if (newParams.length > 0) {
        //curUrl = curUrl.replace(/search\/([^\/]*)\/[^\.]*.html/,"search/$1.html");
        //console.log(1,curUrl,3);return;
        newParams.sort();
        curUrl += '?'+newParams.join('&');
    }
    curUrl = curUrl.replace(/-price-\d*-\d*/, "-price-"+ curPfrom +"-"+ curPto);
    curUrl = curUrl.replace(/\?\&/, "?");
    var curUrllen = curUrl.length;
    if (curUrl.substr(curUrllen-1, 1) == '&') {
        curUrl = curUrl.substr(0, curUrllen-1);
    }
    curUrl = curUrl.replace(/_p-\d+/,"_p-1");
    curUrl = curUrl.replace(/_page\d+/,"_page1");
    //console.log(1,curUrl);return;
    window.location.href = curUrl;
}
